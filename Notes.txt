3) State in Functional Component:-(State is Public) It is nothing but object and we can store data in it.
 Example of State in Functional Component:
 function App() {
  const [data,setData] =useState(0);   
  function updateData()
  {
    setData(data+1);                            /This is the updated value by using State.
  }
  console.warn("-----------------------------------");
  return (
    <div className="App">
      <h1>{data}</h1>
      <button onClick = {updateData}>Update</button>
    </div>


4) State in Class Component:-(State is Public) State is nothing but data container.variable does not update 
the value so we use State.
Example of State in Class Component:-
class App extends Component {

  constructor() {
    super();
    this.state = {
      data: 'Anil'
    }
  }
  apple() {
    this.setState({ data: 'Abhijeet' })
  };
  render() {
    return (
      <div className="App">
        <h1>{this.state.data}</h1>
        <button onClick={() => this.apple()}>Update</button>
      </div>
    );
  }
}
